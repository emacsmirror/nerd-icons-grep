* Configuration


#+begin_src emacs-lisp
(use-package nerd-icons-grep
  :init
  (nerd-icons-grep-mode)
  :custom
  ;; This setting is a pre-requirement, so an icon can be displayed near each
  ;; heading
  (grep-use-headings t)
#+end_src
